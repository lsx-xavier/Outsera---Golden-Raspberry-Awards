    ```ts
      type TriggerProps = {
        text: string;
        icon?: React.ReactNode;
      };

      type OptionItemBySectionProps = {
        textSection?: string; // Serve para quando tiver que separar as opções em seções, bastando apenas passar para o primeiro option da seção.
    ```
